Comandos SQL no SQL Server:

# criar o banco de dados
CREATE DATABASE EstoqueMentoria

# adicionar tabela Insumos
USE Estoque
CREATE TABLE Insumos(
	id_insumo int IDENTITY(1,1),
	nome_insumo varchar(50),
	data_validade date,
	lote int,
	qtde decimal(10,2),
	)

# ver tabela Insumos
SELECT * FROM Insumos
